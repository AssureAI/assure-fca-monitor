+  async function requestRuleGuidance(btn) {
+    const out = btn.nextElementSibling;
+    if (!out) return;
+    try {
+      btn.disabled = true;
+      out.style.display = "block";
+      out.textContent = "Loading…";
+      const payload = {
+        rule_id: btn.getAttribute("data-rule-id") || "",
+        title: btn.getAttribute("data-title") || "",
+        citation: btn.getAttribute("data-citation") || "",
+        decision_logic: btn.dataset.decisionLogic || "",
+        section: btn.getAttribute("data-section") || "",
+        evidence: JSON.parse(btn.getAttribute("data-evidence") || "[]"),
+        fixes: JSON.parse(btn.getAttribute("data-fixes") || "[]"),
+      };
+      const r = await fetch("/llm/rule_guidance", {
+        method: "POST",
+        headers: { "Content-Type": "application/json" },
+        credentials: "same-origin",
+        body: JSON.stringify(payload),
+      });
+      const data = await r.json().catch(() => ({}));
+      if (r.ok && data.guidance) {
+        out.textContent = data.guidance;
+      } else {
+        out.textContent = data.detail || "Guidance request failed.";
+      }
